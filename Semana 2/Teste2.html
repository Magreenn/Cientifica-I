def bissecao(f, a, b, xtol=1e-8, ytol=1e-8):
    niter=0
    while abs(a-b)>xtol and abs(f((a+b)/2))>ytol:
        m = (a+b)/2
        if f(m)*f(a)<0:
            b = m
        else:
            a = m
        niter+=1
    m = (a+b)/2
    return m, niter

def arcoseno(y, xtol=1e-8, ytol=1e-8):
    """Arco seno, via bisseo, dadas as tolerncias (absolutas) em x e em y.
    Retorna um par  (x,n)  tal que  sin(x) ~ y, e onde  n   o nmero de bissees realizadas."""
    def aux(x): return np.sin(x)-y
    return bissecao(aux,-np.pi/2,np.pi/2,xtol,ytol)
